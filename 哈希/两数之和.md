我一开始想用回溯
```javaScript
var twoSum = function(nums, target) {
let path = []
function dfs(start,sum,res){
    if(sum == target && res.length == 2){
        path.push([...res])
        return
    }
    for(let i=start;i<nums.length;i++){
        sum+=nums[i]
        res.push(i)
        dfs(i+1,sum,res)
        sum-=nums[i]
        res.pop()
    }
}
dfs(0,0,[])
return path[0]
};
```
